function parse\_git\_branch {<br />
&nbsp;&nbsp; git branch --no-color 2&gt; /dev/null | sed -e '/^[^\*]/d' -e 's/* \(.*\)/(\1)/' <br />
}<br />

`export PS1="\h:\W \u\$(parse_git_branch)$ "`
